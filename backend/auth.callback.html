(function() { const params = new URLSearchParams(window.location.search); const token = params.get('token'); const id = params.get('id'); const name = params.get('name'); if (!token) { window.location.href = 'auth.html?error=google_failed'; return; } // ✅ Store TEMP (no role yet) localStorage.setItem('tempToken', token); localStorage.setItem('tempId', id); localStorage.setItem('tempName', name); // ✅ Go back to auth page for role selection window.location.href = 'auth.html?google=success'; })();
